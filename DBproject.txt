-- Create independent tables first (no FK dependencies)

-- ADDRESS
CREATE TABLE ADDRESS (
 AddressID  NUMBER(10)    PRIMARY KEY,
 Governorate   VARCHAR2(100)  NOT NULL,
 CityOrCamp   VARCHAR2(100)  NOT NULL,
 AreaOrNeighborhood   VARCHAR2(150),
 StreetDetails   VARCHAR2(200),
 Landmark   VARCHAR2(200) );

-- LOCATION_TYPE
CREATE TABLE LOCATION_TYPE ( 
LocationTypeID  NUMBER(10) PRIMARY KEY, 
TypeName  VARCHAR2(100) NOT NULL ); 

-- ORGANIZATION
CREATE TABLE ORGANIZATION ( 
OrgID NUMBER(10)  PRIMARY KEY,
OrgName  VARCHAR2(150) NOT NULL,
OrgType  VARCHAR2(100),
ContactPhone  VARCHAR2(20),
ContactEmail  VARCHAR2(100) );


-- NEED_TYPE
CREATE TABLE NEED_TYPE (
NeedTypeID  NUMBER(10)  PRIMARY KEY,
NeedName  VARCHAR2(200) NOT NULL,
Category  VARCHAR2(100) );

-- AID_PACKAGE
CREATE TABLE AID_PACKAGE (
PackageID  NUMBER(10)    PRIMARY KEY,
PackageName   VARCHAR2(200) NOT NULL,
PackageCategory  VARCHAR2(100),
UnitName  VARCHAR2(50),
Notes  VARCHAR2(200) );



-- FAMILY (HeadPersonID is NOT NULL)
-- BUT: circular FK with PERSON. We create it NULL TEMPORARILY, then Alter later to add FK + NOT NULL 

CREATE TABLE FAMILY (
FamilyID NUMBER(10)  PRIMARY KEY,
OriginalAddressID  NUMBER(10)  NOT NULL,
HeadPersonID   NUMBER(10), 
PrimaryPhone   VARCHAR2(20) NOT NULL, 
RegistrationDate DATE  NOT NULL, 
TotalMembers   NUMBER(3)  NOT NULL, 
TotalChildren   NUMBER(3) DEFAULT 0,
TotalElderly   NUMBER(3) DEFAULT 0,
TotalPersonsWithDisability NUMBER(3) DEFAULT 0,
TotalOrphans  NUMBER(3) DEFAULT 0,
Status  VARCHAR2(50),
CONSTRAINT fk_family_address FOREIGN KEY (OriginalAddressID)
REFERENCES ADDRESS(AddressID),
CONSTRAINT chk_family_status CHECK (Status IN ('Active','Inactive','Merged')));


--  PERSON
CREATE TABLE PERSON (
PersonID NUMBER(10)    PRIMARY KEY,
FamilyID    NUMBER(10)    NOT NULL,
FullName    VARCHAR2(100) NOT NULL,
Gender VARCHAR2(8) CHECK (Gender IN ('Male','Female')),
DateOfBirth  DATE,
RelationshipToHead VARCHAR2(50),
VulnerabilityFlag CHAR(1) CHECK (VulnerabilityFlag IN ('Y','N')),
Notes VARCHAR2(200),
CONSTRAINT fk_person_family FOREIGN KEY (FamilyID)
REFERENCES FAMILY(FamilyID));

--  LOCATION
CREATE TABLE LOCATION (
LocationID   NUMBER(10)    PRIMARY KEY,
LocationName  VARCHAR2(150) NOT NULL,
LocationTypeID  NUMBER(10)    NOT NULL,
AddressID  NUMBER(10)    NOT NULL,
ManagingOrgID  NUMBER(10),
Capacity  NUMBER(5),
GPS_Lat  NUMBER(10,6),
GPS_Lng  NUMBER(10,6),
CONSTRAINT fk_location_type FOREIGN KEY (LocationTypeID)
REFERENCES LOCATION_TYPE(LocationTypeID),
CONSTRAINT fk_location_address FOREIGN KEY (AddressID)
REFERENCES ADDRESS(AddressID),
CONSTRAINT fk_location_org FOREIGN KEY (ManagingOrgID)
REFERENCES ORGANIZATION(OrgID));

--  STAY
CREATE TABLE STAY ( 
StayID NUMBER(10)  PRIMARY KEY,
FamilyID  NUMBER(10)  NOT NULL,
LocationID  NUMBER(10)  NOT NULL,
StayStartDateTime  DATE  NOT NULL,
StayEndDateTime  DATE,
ShelterUnit    VARCHAR2(50),
IsCurrent  CHAR(1) CHECK (IsCurrent IN ('Y','N')),
CONSTRAINT fk_stay_family FOREIGN KEY (FamilyID)
REFERENCES FAMILY(FamilyID),
CONSTRAINT fk_stay_location FOREIGN KEY (LocationID)
REFERENCES LOCATION(LocationID));

--  DISPLACEMENT_EVENT
CREATE TABLE DISPLACEMENT_EVENT (
EventID    NUMBER(10)   PRIMARY KEY,
FamilyID     NUMBER(10)  NOT NULL,
FromStayID  NUMBER(10),
ToLocationID     NUMBER(10) NOT NULL,
EventStartDateTime  DATE   NOT NULL,
EventType VARCHAR2(50),
CauseTrigger   VARCHAR2(100),
TransportationType  VARCHAR2(200),
CONSTRAINT fk_event_family FOREIGN KEY (FamilyID)
REFERENCES FAMILY(FamilyID),
CONSTRAINT fk_event_fromstay FOREIGN KEY (FromStayID)
REFERENCES STAY(StayID),
CONSTRAINT fk_event_tolocation FOREIGN KEY (ToLocationID)
REFERENCES LOCATION(LocationID));


-- NEED_ASSESSMENT
CREATE TABLE NEED_ASSESSMENT (
AssessmentID NUMBER(10)  PRIMARY KEY,
StayID   NUMBER(10)  NOT NULL,
AssessmentDateTime DATE  NOT NULL,
AssessedByOrgID  NUMBER(10),
Method  VARCHAR2(100),
OverallSeverityScore NUMBER(2)  CHECK (OverallSeverityScore BETWEEN 1 AND 5),
CONSTRAINT fk_assess_stay FOREIGN KEY (StayID)
REFERENCES STAY(StayID),
CONSTRAINT fk_assess_org FOREIGN KEY (AssessedByOrgID)
REFERENCES ORGANIZATION(OrgID));

--  NEED_ITEM
CREATE TABLE NEED_ITEM (
NeedItemID  NUMBER(10)  PRIMARY KEY,
AssessmentID  NUMBER(10)  NOT NULL,
NeedTypeID  NUMBER(10)  NOT NULL,
TargetPersonID  NUMBER(10),
QuantityNeeded  NUMBER(10),
PriorityLevel  NUMBER(1)  CHECK (PriorityLevel BETWEEN 1 AND 5),
Status  VARCHAR2(20),
CONSTRAINT fk_item_assessment FOREIGN KEY (AssessmentID)
REFERENCES NEED_ASSESSMENT(AssessmentID),
CONSTRAINT fk_item_needtype FOREIGN KEY (NeedTypeID)
REFERENCES NEED_TYPE(NeedTypeID),
CONSTRAINT fk_item_person FOREIGN KEY (TargetPersonID)
REFERENCES PERSON(PersonID));


-- AID_DISTRIBUTION
CREATE TABLE AID_DISTRIBUTION (
DistributionID  NUMBER(10)   PRIMARY KEY,
LocationID  NUMBER(10)  NOT NULL,
DonorOrgID  NUMBER(10)  NOT NULL,
PackageID NUMBER(10) NOT NULL,
AssessmentID  NUMBER(10),
DistributionDateTime DATE  NOT NULL,
DistributionNotes  VARCHAR2(200),
CONSTRAINT fk_dist_location FOREIGN KEY (LocationID)
REFERENCES LOCATION(LocationID),
CONSTRAINT fk_dist_org FOREIGN KEY (DonorOrgID)
REFERENCES ORGANIZATION(OrgID),
CONSTRAINT fk_dist_package FOREIGN KEY (PackageID)
REFERENCES AID_PACKAGE(PackageID),
CONSTRAINT fk_dist_assessment FOREIGN KEY (AssessmentID)
REFERENCES NEED_ASSESSMENT(AssessmentID));

-- AID_RECEIPT
CREATE TABLE AID_RECEIPT (
ReceiptID  NUMBER(10) PRIMARY KEY,
DistributionID   NUMBER(10)   NOT NULL,
FamilyID   NUMBER(10)   NOT NULL,
ReceivedDateTime DATE,
Quantity  NUMBER,
ReceiptStatus  VARCHAR2(50),
ProofCode  VARCHAR2(50),
VerifiedByName  VARCHAR2(100),
CONSTRAINT fk_receipt_distribution FOREIGN KEY (DistributionID)
REFERENCES AID_DISTRIBUTION(DistributionID),
CONSTRAINT fk_receipt_family FOREIGN KEY (FamilyID)
REFERENCES FAMILY(FamilyID),
CONSTRAINT uq_receipt UNIQUE (DistributionID, FamilyID));

-- Retrieving data using SELECT with various conditions and queries.
-- 1. Show all families
SELECT * FROM FAMILY;

--2. Display only the active families in the system.
SELECT FamilyID, Status
FROM FAMILY
WHERE Status = 'Active';

--3. List all persons ordered by date of birth (from oldest to youngest).
SELECT FullName, DateOfBirth
FROM PERSON
ORDER BY DateOfBirth;

--4.Where is each family currently staying, and in which shelter unit?
SELECT F.FamilyID, L.LocationName, S.ShelterUnit
FROM FAMILY F
JOIN STAY S ON F.FamilyID = S.FamilyID
JOIN LOCATION L ON S.LocationID = L.LocationID
WHERE S.IsCurrent = 'Y';

--5.How many families are staying in each location?
SELECT L.LocationName, COUNT(*) AS FamiliesCount
FROM STAY S
JOIN LOCATION L ON S.LocationID = L.LocationID
GROUP BY L.LocationName;

6. What are the high-priority needs identified for displaced families?
SELECT NT.NeedName, NI.PriorityLevel
FROM NEED_ITEM NI
JOIN NEED_TYPE NT ON NI.NeedTypeID = NT.NeedTypeID
WHERE NI.PriorityLevel >= 4;

7. Which aid package were recived by each family and when were they distributed?
SELECT F.FamilyID, P.PackageName, R.ReceiptStatus, D.DistributionDateTime
FROM AID_RECEIPT R
JOIN FAMILY F ON R.FamilyID = F.FamilyID
JOIN AID_DISTRIBUTION D ON R.DistributionID = D.DistributionID
JOIN AID_PACKAGE P ON D.PackageID = P.PackageID;
